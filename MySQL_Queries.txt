-- Display the full name of customers (use CONCAT) and the total amount they have spent across all orders, ordered from highest spender to lowest.

GROUP BY c.id, c.first_name, c.last_name

-- Display orders along with customer full name and restaurant name, sorted by order date descending, but limit results to 5 records and skip the first 5 (OFFSET).
-- no 10 query 
SELECT o.id, CONCAT(c.first_name, ' ' , c.last_name) as full_name , r.name 
FROM food_delivery.`order` as o 
JOIN food_delivery.restaurant as r on r.id = o.restaurant_id 
JOIN food_delivery.customer as c on c.id = o.customer_id 
order by o.order_date DESC LIMIT 5 OFFSET 5 ;

-- List all orders that were paid using either 'cash' or 'card' (use IN), including customer and payment details
-- no 15 query 
SELECT o.id, CONCAT(c.first_name, ' ' , c.last_name) as full_name , p.payment_method, p.payment_status 
from food_delivery.`order` as o 
JOIN food_delivery.customer as c on c.id = o.customer_id 
JOIN food_delivery.payment as p on o.id = p.order_id 
WHERE  p.payment_method IN ('cash', 'card');
